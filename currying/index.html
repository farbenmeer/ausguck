Haskell Curry

function add(a, b) {
    return a + b;
}

function add(a) {
    return function(b) {
        return a + b;
    }
}

add(1)(2); // 3

const add1 = add(1);
add1(2); // 3

lodash.curry

import {curry} from 'lodash.curry';

const abc = (a, b, c) => {
    return [a, b, c];
};

const curried = curry(abc);

curried(1)(2)(3);
// => [1, 2, 3]
 
curried(1, 2)(3);
// => [1, 2, 3]
 
curried(1, 2, 3);
// => [1, 2, 3]

If you apply the same arguments to a function more than once you should think if currying would be a good choice for you.

Some languages that are curried by default: Haskell, Elm, OCaml, F# ...

How does currying looks like in other languages?

Go:

func add(x int) func(int) int {
    return func(y int) int {
        return x + y
    }
}

func main() {
    fmt.Println(add(1)(2)) // 3
    add1 := add(1)
    fmt.Println(add1(2)) // 3
    fmt.Println(add1(3)) // 4
}
