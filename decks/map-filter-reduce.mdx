map 

---

```js
var new_array = arr.map(function callback(currentValue[, index[, array]]) {
    // Return element for new_array
}[, thisArg])
```

---

```js
[1, 2, 3].map(i => i * 2)
```

---

```js
[1, 2, 3].map(i => i * 2) // [2, 4, 6]
```

```js
var newArray = arr.filter(callback(element[, index[, array]])[, thisArg])
```

---

reduce

---

```js
arr.reduce(callback[, initialValue])
```

---

callback:

```
callback(accumulator, currentValue, currentIndex, array)
```

---

```js
[1, 2, 3].reduce((acc, curr) => acc += curr, 0)
```

---

map:

---

map:

```js
[1, 2, 3].reduce((acc, curr) => acc.concat(curr * 2) , [])
```
---

map:

```js
[1, 2, 3].reduce((acc, curr) => acc.concat(curr * 2) , []) // 2, 4, 6
```

---

```js
const myMap = (arr, fn) => arr.reduce((acc, curr) => acc.concat(fn(curr)), [])
```

---

```js
const arr = [1, 2, 3]
const fn = i => i * 2
const result = myMap(arr, fn)

// result === [2, 4, 6]
```

---

filter:

---

filter:

```js
const myFilter = (arr, fn) => arr.reduce((acc, curr) => fn(curr) ? acc.concat(curr) : acc , [])
```

---

```js
const arr = [1, 2, 3]
const fn = i => i < 3
const result = myFilter(arr, fn)

// result === [1, 2]
```
