Asynchronous JavaScript, or JavaScript that uses callbacks, is hard to get right intuitively. A lot of code ends up looking like this:

setTimeout(() => {
  console.log('1sec passed')
  setTimeout(() => {
    console.log('2sec passed')
    setTimeout(() => {
      console.log('3sec passed')
      setTimeout(() => {
        console.log('4sec passed')
        setTimeout(() => {
          console.log('4sec passed')
        }, 1000)
      }, 1000)
    }, 1000)
  }, 1000)
}, 1000)

Promises were introduced to help!

The Promise object represents the eventual completion (or failure) of an asynchronous operation, and its resulting value.

Creating a promise

const myPromise = new Promise(function(resolve, reject) {

});

// => { /* promise object */ }

Resolving a promise

const myPromise = new Promise(function(resolve, reject) {
  resolve('fooo')
});

myPromise.then((result) => {
  console.log(result)
})

Rejecting a promise

const myPromise = new Promise(function(resolve, reject) {
  reject(new Error('omg something went wrong'))
});

myPromise.catch((error) => {
  console.log('error')
})

// => { /* promise object */ }

Converting a callback function to an promise (async) function

const fs = require('fs')

function readFileAsync(argument) {
  return new Promise((resolve, reject) => {
    fs.readFile('/etc/passwd', (err, data) => {
      if (err) {
        return reject(error)
      } else {
        return resolve(data)
      }
    });
  })
}

readFileAsync('./somepath')
  .then((data) => {
    console.log(data)
  })
  .catch((error) => {
    console.error(error)
  })

promisify

function promisify (targetFunc) {
  return (...args) => {
    return new Promise((resolve, reject) => {
      targetFunc(...args, (error, result) => {
        if (error) {
          error(error)
        }
        resolve(result)
      })
    })
  }

Promises flatten itself

Promise.resolve().then(() => {
  return Promise.resolve().then(() => {
    return 'foo'
  })
})

// => 'foo'

Async Await

async function foo () {
  return 'foo'
}

same as

function foo () {
  return new Promise((resolve, reject) => {
    resolve('foo')
  })
}

Implementing a sleep function

function sleep(ms) {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve()
    }, ms)
  })
}

Using a sleep function

(async () => {
  console.log('1sec passed')
  await sleep(1000)
  console.log('2sec passed')
  await sleep(1000)
  console.log('3sec passed')
  await sleep(1000)
  console.log('4sec passed')
  await sleep(1000)
  console.log('5sec passed')
  await sleep(1000)
})()

Good to know

const {promisify} = require('util');
const fs = require('fs');
const readFileAsync = promisify(fs.readFile);

(async () => {
  try {
    var data = await readFileAsync('./path', 'utf8')
  } catch (error) {
    console.error(error)
  }

  console.log(data)
})()

Future

const fs = require('fs')

(async () => {
  var data = await fs.promise.readFile('./path', 'utf8')
  console.log(data)
})()

Bad things about promises

  • Promises are a bit slower than callbacks.
  • You can`t cancel a promise
  • Promise can only resolve once
